@startuml

class build {
	bd_id : int
	bd_ac_id : int #
	bd_position : int [2]
	bd_title : varchar [32]
	bd_password : varchar [25]
	bd_kind : varchar [32]
}

class chest {
	ch_id : int
	ch_bd_id : int #
	ch_name : varchar [32]
}

class chest_object {
	cho_id : int
	cho_ch_id : int #
	cho_name : varchar [32]
	cho_count : int16_t
}

hide build methods
hide chest methods
hide chest_object methods

build "1" -- "1" chest
chest "1" -- "0..*" chest_object

@enduml
