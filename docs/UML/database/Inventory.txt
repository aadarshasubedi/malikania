	invo_name : varchar [32]
	invo_count : int16_t
}

}

	atf_id : int
	atf_ch_id : int #
	atf_name : varchar [32]
}

class quest {
	qs_id : int
	qs_ch_id : int
	qs_name : varchar [32]
	qs_status : enum
}
